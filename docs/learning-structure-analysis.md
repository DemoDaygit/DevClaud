      }

      item.repetitions++;
    }

    // –û–±–Ω–æ–≤–∏—Ç—å easeFactor
    item.easeFactor = item.easeFactor + (0.1 - (5 - quality) * (0.08 + (5 - quality) * 0.02));

    if (item.easeFactor < 1.3) {
      item.easeFactor = 1.3;
    }

    item.nextReview = new Date(Date.now() + item.interval * 24 * 60 * 60 * 1000);

    return item;
  }
}
